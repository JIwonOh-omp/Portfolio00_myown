```

├── preview1.html           # 컬러 미리보기 - 하늘색+연노랑
├── preview2.html           # 컬러 미리보기 - 초록+연분홍

└── assets/
    ├── css/
    │   └── style.css       # 메인 스타일시트 (커스터마이징)
    ├── img/
    │   ├── profile-img-jiwon.jpg   # 프로필 사진
    │   └── portfolio/              # 포트폴리오 썸네일 (실제 스크린샷)
    ├── js/
    │   └── main.js
    └── vendor/             # 외부 라이브러리
```
